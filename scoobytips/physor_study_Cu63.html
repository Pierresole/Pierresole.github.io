def print_attributes_and_methods(obj):
    print("Attributes and Methods of", type(obj))
    attributes = []
    methods = []
    for name in dir(obj):
        # Filter out special methods/attributes
        if not name.startswith('__'):
            if callable(getattr(obj, name)):
                methods.append(name)
            else:
                attributes.append(name)
    
    print("\nAttributes:")
    for attr in attributes:
        print(attr)
    
    print("\nMethods:")
    for method in methods:
        print(method)

def plot_data_from_file(file_path):
    # Load ACE data from the specified file
    nuclei = openmc.data.IncidentNeutron.from_ace(file_path)
    reaction = nuclei[2]
    print(reaction)
    
    # Assuming one single temperature
    first_temperature_key = list(nuclei.energy.keys())[0]  # Get the first key of the dictionary
    energies = nuclei.energy[first_temperature_key]  # Use the first key to get the energies instead of nuclei.energy['0K']
    reaction_xs = reaction.xs[first_temperature_key](energies)
    # Plot the data
    plt.loglog(energies, reaction_xs, label=file_path.split('/')[-1])  # Use the file name as the label
